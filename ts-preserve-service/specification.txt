/**
* The preserve method in PreserveServiceImpl.java is excessively long and contains
* numerous method calls that invoke other services. To improve readability, the
* specification has been divided into two categories: ENDPOINT and SERVICE METHOD.
*/

SERVICE: Preserve Service

ENDPOINT: GET api/v1/preserveservice/welcome
IF request is GET api/v1/preserveservice/welcome THEN
    RETURN welcoming message
ENDIF


ENDPOINT: POST api/v1/preserveservice/preserve
IF request is POST api/v1/preserveservice/preserve THEN
    RECEIVE OrderTicketsInfo from request body
    CALL preserve method in PreserveService with OrderTicketsInfo
    RETURN result of preserve method
ENDIF


SERVICE METHOD: preserve in PreserveService
IF method preserve is called THEN
    RECEIVE OrderTicketsInfo
    CALL checkSecurity method with OrderTicketsInfo
    CALL getContactsById method with contactsId from OrderTicketsInfo
    CALCULATE TripAlLDetailInfo
    CALL getTripAllDetailInformation method with TripAllDetailInfo
    CALL queryForStationId method with from location from OrderTicketsInfo
    CALL queryForStationId method with to location OrderTicketsInfo
    CREATE new Order with data from body
    CALCULATE Travel
    REQUEST TravelResult via POST /api/v1/ticketinfoservice/ticketinfo with Travel in body
    CALL dispatchSeat method with Date from OrderTicketsInfo,
                trainNumber from Order,
                fromStationId from result of queryForStationId method,
                toStationId from result of queryForStationId method
    CALL createOrder method with Order
    CALL addAssuranceForOrder method with Assurance from OrderTicketInfo and Id from result of createOrder method
    CALCULATE FoodOrder from OrderTicketsInfo
    CALL createFoodOrder method with FoodOrder
    CALCULATE Consign from result of createOrder method and from OrderTicketsInfo
    CALL createConsign method with Consign
    CALL getAccount method with accountId from Order
    CALCULATE NotifyInfo from result of getAccount method and Order
    CALL sendEmail method with NotifyInfo
    IF result of all calls successful
        RETURN successful
    ELSE
        RETURN failure
ENDIF


SERVICE METHOD: checkSecurity in PreserveService
IF method checkSecurity is called THEN
    RECEIVE accountId
    REQUEST CheckResult via GET /api/v1/securityservice/securityConfigs/{accountId} with accountId in paths
    RETURN response body of reCheckResult
ENDIF
