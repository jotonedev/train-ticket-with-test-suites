SERVICE: Order Other Service

ENDPOINT: GET api/v1/orderOtherService/welcome
IF request is GET api/v1/orderOtherService/welcome THEN
    RETURN welcoming message
ENDIF


ENDPOINT: POST api/v1/orderOtherService/orderOther/tickets
IF request is POST api/v1/orderOtherService/orderOther/tickets THEN
    RECEIVE Seat request from request body
    FIND sold tickets for the given seat request
    RETURN sold tickets and success
ENDIF


ENDPOINT: POST api/v1/orderOtherService/orderOther
IF request is POST api/v1/orderOtherService/orderOther THEN
    RECEIVE Order from request body
    FIND Order with received id in database
    IF Order exists
        RETURN already exists
    ELSE
        CREATE new Order with data from body
        STORE created Order in database
        RETURN created Order and success
    ENDIF
ENDIF
