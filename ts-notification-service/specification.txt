    RECEIVE NotifyInfo object
    IF NotifyInfo object is valid THEN
        CALL preserveSuccess function with NotifyInfo object
        // Inside preserveSuccess function
        CREATE Mail object with details from NotifyInfo
        SET mail subject to "Preserve Success"
        SET mail template to "preserve_success.ftl"
