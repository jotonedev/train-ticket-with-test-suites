SERVICE: Food Map Service

ENDPOINT: GET /api/v1/foodmapservice/foodstores/welcome
IF request is GET /api/v1/foodmapservice/foodstores/welcome THEN
    RETURN "Welcome to [ Food store Service ] !"
ENDIF

ENDPOINT: GET /api/v1/foodmapservice/foodstores
IF request is GET /api/v1/foodmapservice/foodstores THEN
    CALL listFoodStores function
    RETURN list of all food stores
    IF list has at least one element
        RETURN list and success
    ELSE
        RETURN no content
    ENDIF
ENDIF

ENDPOINT: GET /api/v1/foodmapservice/foodstores/{stationId}
IF request is GET /api/v1/foodmapservice/foodstores/{stationId} THEN
    RECEIVE Station ID
    CALL listFoodStoresByStationId function with Station ID
    RETURN list of food stores for the specified station
    IF list has at least one element
        RETURN list and success
    ELSE
        RETURN no content
    ENDIF
ENDIF

ENDPOINT: POST /api/v1/foodmapservice/foodstores
IF request is POST /api/v1/foodmapservice/foodstores THEN
    RECEIVE list of Station IDs
    CALL getFoodStoresByStationIds function with list of Station IDs
    RETURN list of food stores for the specified stations
    IF list has at least one element
        RETURN list and success
    ELSE
        RETURN no content
    ENDIF
ENDIF
